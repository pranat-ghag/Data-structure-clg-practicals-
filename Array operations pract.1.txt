#include <iostream.h>
#include <conio.h>

void traverse(int arr[], int n) {
    int i;
    cout << "Array elements:\n";
    for (i = 0; i < n; i++) {
	cout << arr[i] << " ";
    }
    cout << "\n";
}

void insert(int arr[], int &n) {
    int pos, val, i;
    cout << "Enter position to insert (0 to " << n << "): ";
    cin >> pos;
    if (pos < 0 || pos > n) {
	cout << "Invalid position!\n";
	return;
    }
    cout << "Enter value to insert: ";
    cin >> val;
    for (i = n; i > pos; i--) {
	arr[i] = arr[i - 1];
    }
    arr[pos] = val;
    n++;
    cout << "Element inserted.\n";
}

void deleteElement(int arr[], int &n) {
    int pos, i;
    cout << "Enter position to delete (0 to " << n - 1 << "): ";
    cin >> pos;
    if (pos < 0 || pos >= n) {
	cout << "Invalid position!\n";
	return;
    }
    for (i = pos; i < n - 1; i++) {
	arr[i] = arr[i + 1];
    }
    n--;
    cout << "Element deleted.\n";
}

void sortArray(int arr[], int n) {
    int i, j, temp;
    for (i = 0; i < n - 1; i++) {
	for (j = 0; j < n - i - 1; j++) {
	    if (arr[j] > arr[j + 1]) {
		temp = arr[j];
		arr[j] = arr[j + 1];
		arr[j + 1] = temp;
	    }
	}
    }
    cout << "Array sorted.\n";
}

void mergeArrays(int arr[], int &n) {
    int arr2[50], n2, i;
    cout << "Enter number of elements in second array: ";
    cin >> n2;
    if (n + n2 > 100) {
	cout << "Merging would exceed array limit.\n";
	return;
    }
    cout << "Enter elements of second array:\n";
    for (i = 0; i < n2; i++) {
	cin >> arr2[i];
    }
    for (i = 0; i < n2; i++) {
	arr[n] = arr2[i];
	n++;
    }
    cout << "Arrays merged.\n";
}

void main() {
    clrscr();
    int arr[100], n, choice;
    cout << "Enter number of elements: ";
    cin >> n;
    cout << "Enter elements:\n";
    for (int i = 0; i < n; i++) {
	cin >> arr[i];
    }
    do {
	cout << "\n--- MENU ---\n";
	cout << "1. Traverse\n";
	cout << "2. Insert\n";
	cout << "3. Delete\n";
	cout << "4. Sort\n";
	cout << "5. Merge\n";
	cout << "6. Exit\n";
	cout << "Enter your choice: ";
	cin >> choice;
	switch (choice) {
	    case 1:
		traverse(arr, n);
		break;
	    case 2:
		insert(arr, n);
		break;
	    case 3:
		deleteElement(arr, n);
		break;
	    case 4:
		sortArray(arr, n);
		break;
	    case 5:
		mergeArrays(arr, n);
		break;
	    case 6:
		cout << "Exiting...\n";
		break;
	    default:
		cout << "Invalid choice!\n";
	}
    } while (choice != 6);
    getch();
}
